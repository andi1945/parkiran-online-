            // Fungsi untuk mengambil data dari localStorage (Memastikan data tidak hilang saat refresh)
            function getParkingData() {
                return JSON.parse(localStorage.getItem(localStorageKey)) || {};
            }

            // Fungsi untuk menyimpan data ke localStorage
            function saveParkingData(data) {
                localStorage.setItem(localStorageKey, JSON.stringify(data));
            }

            // Fungsi untuk membuat layout parkir
            function createParkingLayout() {
                layout.innerHTML = ''; // Kosongkan layout
                const parkingData = getParkingData();

                for (let i = 1; i <= totalSpots; i++) {
                    const spotId = `P-${String(i).padStart(2, '0')}`; // Format P-01, P-02, dst.
                    const spot = document.createElement('div');
                    spot.classList.add('spot');
                    spot.id = spotId;

                    const spotData = parkingData[spotId];

                    if (spotData) {
                        // Jika terisi
                        spot.classList.add('occupied');
                        spot.innerHTML = `

                        `;
                    } else {
                        // Jika kosong
                        spot.classList.add('available');
                        spot.innerHTML = `<span>${spotId}</span>`;
                    }
                    layout.appendChild(spot);
                }
            }

            // Fungsi untuk memperbarui status (tersedia/terisi)
            function updateStatus() {
                const parkingData = getParkingData();
                const occupiedCount = Object.keys(parkingData).length;
                const availableCount = totalSpots - occupiedCount;

                availableEl.textContent = availableCount;
                occupiedEl.textContent = occupiedCount;
            }

            // Fungsi utama untuk me-render seluruh aplikasi
            function renderApp() {
                createParkingLayout();
                updateStatus();
